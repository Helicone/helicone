    ```bash
    curl --request POST \
        --url https://oai.helicone.ai/v1/chat/completions \
        --header 'Authorization: Bearer <<YOUR_OPENAI_API_KEY>>' \
        --header 'Content-Type: application/json' \
        --header 'Helicone-Auth: Bearer <<YOUR_HELICONE_API_KEY>>' \
        --data '{
            "model": "gpt-3.5-turbo",
            "messages": [
                {
                    "role": "system",
                    "content": "Say Hello!"
                }
            ],
            "temperature": 1,
            "max_tokens": 10
    }'
    ```
